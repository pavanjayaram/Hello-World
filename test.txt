import React from 'react';
import { ApiError } from '../services/apiClient';

interface ErrorDisplayProps {
  error: string | Error | ApiError | null;
  title?: string;
  showRetry?: boolean;
  onRetry?: () => void;
  className?: string;
}

const ErrorDisplay: React.FC<ErrorDisplayProps> = ({
  error,
  title = "Error",
  showRetry = false,
  onRetry,
  className = ""
}) => {
  if (!error) return null;

  const getErrorMessage = (): string => {
    if (typeof error === 'string') {
      return error;
    }
    if (error instanceof ApiError) {
      return error.message;
    }
    if (error instanceof Error) {
      return error.message;
    }
    return 'An unknown error occurred';
  };

  const getErrorCode = (): string | undefined => {
    if (error instanceof ApiError) {
      return error.errorCode;
    }
    return undefined;
  };

  const getErrorDetails = (): Record<string, any> | undefined => {
    if (error instanceof ApiError) {
      return error.details;
    }
    return undefined;
  };

  const getTroubleshootingSteps = (): string[] => {
    if (error instanceof ApiError) {
      const details = error.details;
      
      // LLM unavailability errors
      if (error.isErrorCode('LLM_UNAVAILABLE') || error.isErrorCode('LLM_CONNECTION_ERROR')) {
        return details?.troubleshooting || [
          "Please start Ollama with 'ollama serve'",
          "Ensure codellama:7b is pulled with 'ollama pull codellama:7b'",
          "Check if the model endpoint is accessible",
          "Verify firewall settings allow connection to the model endpoint"
        ];
      }
      
      // Network errors
      if (error.isNetworkError()) {
        return [
          "Check your internet connection",
          "Verify the backend server is running",
          "Check if the API endpoint is accessible",
          "Try refreshing the page"
        ];
      }
      
      // Server errors
      if (error.isServerError()) {
        return [
          "The server encountered an internal error",
          "Please try again in a few moments",
          "If the problem persists, contact support"
        ];
      }
      
      // Configuration validation errors
      if (error.isErrorCode('INVALID_CONFIG')) {
        return [
          "Review your configuration for missing or invalid fields",
          "Ensure all required fields are filled out",
          "Check that file paths and connection strings are correct",
          "Verify delimiter settings for delimited files"
        ];
      }
    }
    
    return [];
  };

  const errorCode = getErrorCode();
  const errorDetails = getErrorDetails();
  const troubleshootingSteps = getTroubleshootingSteps();

  return (
    <div className={`error-display ${className}`}>
      <div className="error-header">
        <h4 className="error-title">{title}</h4>
        {errorCode && (
          <span className="error-code">Error Code: {errorCode}</span>
        )}
      </div>
      
      <div className="error-content">
        <p className="error-message">{getErrorMessage()}</p>
        
        {/* Error Details */}
        {errorDetails && Object.keys(errorDetails).length > 0 && (
          <div className="error-details">
            <h5>Details:</h5>
            <div className="error-details-content">
              {errorDetails.errors && Array.isArray(errorDetails.errors) && (
                <div className="error-list">
                  <strong>Validation Errors:</strong>
                  <ul>
                    {errorDetails.errors.map((err: string, idx: number) => (
                      <li key={idx}>{err}</li>
                    ))}
                  </ul>
                </div>
              )}
              
              {errorDetails.warnings && Array.isArray(errorDetails.warnings) && (
                <div className="warning-list">
                  <strong>Warnings:</strong>
                  <ul>
                    {errorDetails.warnings.map((warning: string, idx: number) => (
                      <li key={idx}>{warning}</li>
                    ))}
                  </ul>
                </div>
              )}
              
              {errorDetails.endpoint && (
                <div className="error-endpoint">
                  <strong>Endpoint:</strong> <code>{errorDetails.endpoint}</code>
                </div>
              )}
              
              {errorDetails.request_id && (
                <div className="error-request-id">
                  <strong>Request ID:</strong> <code>{errorDetails.request_id}</code>
                </div>
              )}
            </div>
          </div>
        )}
        
        {/* Troubleshooting Steps */}
        {troubleshootingSteps.length > 0 && (
          <div className="troubleshooting">
            <h5>Troubleshooting:</h5>
            <ul className="troubleshooting-list">
              {troubleshootingSteps.map((step, idx) => (
                <li key={idx}>{step}</li>
              ))}
            </ul>
          </div>
        )}
        
        {/* Retry Button */}
        {showRetry && onRetry && (
          <div className="error-actions">
            <button
              onClick={onRetry}
              className="btn btn-primary retry-button"
            >
              Try Again
            </button>
          </div>
        )}
      </div>
    </div>
  );
};

export default ErrorDisplay;
