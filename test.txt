
            
            # Support both connection (registry) and connectionString (direct)
            if 'connection' in target_data:
                config['connection'] = target_data['connection']
            if 'connectionString' in target_data:
                config['connectionString'] = target_data['connectionString']
                
            # Add optional DB properties
            for key in ['schema', 'primary_key', 'batchsize', 'isolationLevel']:
                if key in target_data:
                    config[key] = target_data[key]
                    
            return config
