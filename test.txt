def build_llm_prompt(config: Dict[str, Any]) -> str:
    config_json = json.dumps(config, indent=2)
    return f"""
You are an expert Data Engineer specializing in Python, Polars, and Oracle databases.

Based on the following JSON ETL pipeline configuration, generate a complete, independent, and runnable Python script using the `polars` library.

The script MUST:
1. Use `polars` for all data transformations, reading, and writing. Do NOT use PySpark or Pandas.
2. Read data from the configured source (CSV, Parquet, or Database).
3. Apply all listed field and row transformations step-by-step.
4. Write the final DataFrame to the configured target (usually Oracle Database).
5. Handle parallelism properly. For Polars, this means utilizing LazyFrames (`.lazy()`) and `collect(streaming=True)` if appropriate for large datasets (up to 100 million rows).
6. For Oracle database reads/writes, use highly optimized concurrent logic utilizing Python multiprocessing or optimized batch configurations if `polars.write_database` is not sufficient natively.
7. Include robust error handling and logging.
8. Be well-commented and clean.
9. ONLY output the Python code, surrounded by ```python ... ``` blocks. Do not include markdown explanations outside the code block.

Pipeline Configuration JSON:
{config_json}
"""

@app.route('/api/generate-ollama-code', methods=['POST'])
def generate_ollama_code():
    try:
        config = request.json
        prompt = build_llm_prompt(config)
        
        import requests
        
        # Call local Ollama
        response = requests.post('http://localhost:11434/api/generate', json={
            "model": "codellama:7b",
            "prompt": prompt,
            "stream": False
        }, timeout=300)
        
        response.raise_for_status()
        result = response.json()
        
        # Extract the code from the response
        code = result.get('response', '')
        
        # Clean markdown code blocks if present
        if code.strip().startswith('```python'):
            code = code.strip().split('```python', 1)[1]
            if '```' in code:
                code = code.rsplit('```', 1)[0]
        elif code.strip().startswith('```'):
            code = code.strip().split('```', 1)[1]
            if '```' in code:
                code = code.rsplit('```', 1)[0]
                
        return jsonify({
            "success": True,
            "code": code.strip(),
            "status": "success"
        })
    except requests.exceptions.ConnectionError:
        return jsonify({
            "success": False,
            "error": "Failed to connect to Ollama. Make sure Ollama is running locally on port 11434 (`ollama serve`)."
        }), 500
    except Exception as e:
        return jsonify({
            "success": False,
            "error": str(e)
        }), 500


@app.route('/api/generate-gemini-code', methods=['POST'])
def generate_gemini_code():
    try:
        config = request.json
        prompt = build_llm_prompt(config)
        
        gemini_api_key = os.environ.get('GEMINI_API_KEY')
        if not gemini_api_key:
            return jsonify({
                "success": False,
                "error": "GEMINI_API_KEY environment variable is not set in the backend .env file."
            }), 400
            
        import requests
        url = f"https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key={gemini_api_key}"
        
        response = requests.post(url, json={
            "contents": [{"parts": [{"text": prompt}]}]
        }, timeout=300)
        
        response.raise_for_status()
        result = response.json()
        
        try:
            code = result.get('candidates', [])[0].get('content', {}).get('parts', [])[0].get('text', '')
        except (IndexError, AttributeError):
            code = f"# Failed to parse Gemini response:\n# {json.dumps(result, indent=2)}"
        
        # Clean markdown code blocks if present
        if code.strip().startswith('```python'):
            code = code.strip().split('```python', 1)[1]
            if '```' in code:
                code = code.rsplit('```', 1)[0]
        elif code.strip().startswith('```'):
            code = code.strip().split('```', 1)[1]
            if '```' in code:
                code = code.rsplit('```', 1)[0]
                
        return jsonify({
            "success": True,
            "code": code.strip(),
            "status": "success"
        })
    except Exception as e:
        # Sanitize error message to remove any API key
        error_msg = str(e)
        if gemini_api_key:
            error_msg = error_msg.replace(gemini_api_key, '***REDACTED***')
        return jsonify({
            "success": False,
            "error": error_msg
        }), 500
