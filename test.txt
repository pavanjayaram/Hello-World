def _parse_oracle_connection(cs: str):
    """Parse Oracle connection string into (username, password, host, port, service_name).
    Handles formats:
      - oracle://user:pass@host:1521/SERVICE
      - oracle+cx_oracle://user:pass@host:1521/SERVICE
      - jdbc:oracle:thin:user/pass@host:1521:SID
      - jdbc:oracle:thin:@//host:1521/SERVICE
      - jdbc:oracle:thin:@host:1521:SID
    """
    import re
    from urllib.parse import urlparse, unquote

    if cs.startswith('jdbc:oracle:thin:'):
        after = cs[len('jdbc:oracle:thin:'):]
        m = re.match(r'^([^@/]+)/([^@]+)@(.+)$', after)
        if m:
            uname, pwd, rest = m.group(1), m.group(2), m.group(3)
        else:
            uname, pwd = '', ''
            rest = after.lstrip('@')

        if rest.startswith('//'):
            inner = rest[2:]
            host_port, _, svc = inner.partition('/')
            host, _, port_str = host_port.partition(':')
            port = int(port_str) if port_str.isdigit() else 1521
        else:
            parts = rest.split(':')
            host = parts[0] if len(parts) > 0 else 'localhost'
            port = int(parts[1]) if len(parts) > 1 and parts[1].isdigit() else 1521
            svc = parts[2] if len(parts) > 2 else ''

        return uname, pwd, host, port, svc

    normalised = re.sub(r'^oracle\+[a-z_]+://', 'oracle://', cs)
    if not normalised.startswith('oracle://'):
        normalised = 'oracle://' + normalised.lstrip('/')

    # oracle://user:pass@//host:port/SERVICE  →  oracle://user:pass@host:port/SERVICE
    # The double-slash after @ is valid Oracle syntax but breaks urlparse
    normalised = normalised.replace('@//', '@')

    parsed = urlparse(normalised)
    uname = unquote(parsed.username or '')
    pwd   = unquote(parsed.password or '')
    host  = parsed.hostname or 'localhost'
    port  = parsed.port or 1521
    svc   = parsed.path.lstrip('/') if parsed.path else ''
    return uname, pwd, host, port, svc


class PipelineConfigGenerator:
    """Generates ETL pipeline JSON configurations from frontend requests"""
    
    @staticmethod
    def generate_pipeline_config(data: Dict[str, Any]) -> Dict[str, Any]:
        """
        Generate complete pipeline configuration from frontend data
        
        Args:
            data: Frontend form data containing pipeline details
            
        Returns:
            Complete pipeline JSON configuration
        """
        config = {
            "pipeline": PipelineConfigGenerator._build_pipeline_section(data),
            "source": PipelineConfigGenerator._build_source_section(data.get('source', {})),
            "transformations": PipelineConfigGenerator._build_transformations(data.get('transformations', [])),
            "target": PipelineConfigGenerator._build_target_section(data.get('target', {})),
            "audit": PipelineConfigGenerator._build_audit_section(data.get('audit', {})),
            "reject": PipelineConfigGenerator._build_reject_section(data.get('reject', {}))
        }
        
        return config
    
    @staticmethod
    def _build_pipeline_section(data: Dict[str, Any]) -> Dict[str, Any]:
        """Build pipeline metadata section"""
        pipeline_data = data.get('pipeline', {})
        return {
            "pipeline_id": pipeline_data.get('pipeline_id', f"PIPELINE_{datetime.now().strftime('%Y%m%d_%H%M%S')}"),
            "pipeline_type": pipeline_data.get('pipeline_type', 'FILE_TO_FILE'),
            "execution_mode": pipeline_data.get('execution_mode', 'BATCH'),
            "run_date": pipeline_data.get('run_date', datetime.now().strftime('%Y-%m-%d')),
            "version": pipeline_data.get('version', '1.0'),
            "log_level": pipeline_data.get('log_level', 'INFO')
        }
    
    @staticmethod
    def _build_source_section(source_data: Dict[str, Any]) -> Dict[str, Any]:
        """Build source configuration section"""
        source_type = source_data.get('type', 'FILE')
        
        if source_type == 'FILE':
            return {
                "type": "FILE",
                "format": source_data.get('format', 'csv'),
                "path": source_data.get('path', ''),
                "options": source_data.get('options', {})
            }
        else:  # DB
            return {
                "type": "DB",
                "db_type": source_data.get('db_type', 'postgres'),
                "table": source_data.get('table', ''),
                "connection": source_data.get('connection', 'pg_conn')
            }
    
    @staticmethod
    def _build_transformations(transformations: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
        """Build transformations array"""
        result = []
        
        for transform in transformations:
            transform_type = transform.get('type')
            
            if transform_type == 'column':
                result.append({
                    "type": "column",
                    "action": transform.get('action'),
                    "column": transform.get('column'),
                    **{k: v for k, v in transform.items() 
                       if k not in ['type', 'action', 'column'] and v is not None}
                })
            
            elif transform_type == 'row':
                result.append({
                    "type": "row",
                    "action": transform.get('action'),
                    **{k: v for k, v in transform.items() 
                       if k not in ['type', 'action'] and v is not None}
                })
            
            elif transform_type == 'lookup':
                lookup_config = {
                    "type": "lookup",
                    "lookup_table": transform.get('lookup_table'),
                    "source_type": transform.get('source_type', 'FILE'),
                    "path": transform.get('path'),
                    "join_keys": transform.get('join_keys', {}),
                    "join_type": transform.get('join_type', 'left'),
                    "columns": transform.get('columns', []),
                    "on_missing": transform.get('on_missing', 'default'),
                    "broadcast": transform.get('broadcast', False)
                }
                
                # Add conditional fields based on source type
                if transform.get('source_type') == 'DB':
                    lookup_config['db_type'] = transform.get('db_type')
                    # Map frontend 'connection_string' to backend 'connectionString' if needed, 
                    # or keep consistent. Backend uses 'connectionString' in ad-hoc loading.
                    lookup_config['connection_string'] = transform.get('connection_string')
                else:
                    lookup_config['file_format'] = transform.get('file_format', 'csv')
                    
                result.append(lookup_config)
            
            elif transform_type == 'cdc':
                result.append({
                    "type": "cdc",
                    "mode": transform.get('mode', 'SCD_TYPE_2'),
                    "business_keys": transform.get('business_keys', []),
                    "compare_columns": transform.get('compare_columns', 'ALL'),
                    "effective_start": transform.get('effective_start', 'eff_start_dt'),
                    "effective_end": transform.get('effective_end', 'eff_end_dt'),
                    "current_flag": transform.get('current_flag', 'is_current'),
                    "operation_flag": transform.get('operation_flag', 'op_type')
                })
        
        return result
    
    @staticmethod
    def _build_target_section(target_data: Dict[str, Any]) -> Dict[str, Any]:
        """Build target configuration section"""
        target_type = target_data.get('type', 'FILE')
        
        if target_type == 'FILE':
            config = {
                "type": "FILE",
                "format": target_data.get('format', 'parquet'),
                "path": target_data.get('path', ''),
                "mode": target_data.get('mode', 'overwrite')
            }
            
            if target_data.get('partition_by'):
                config['partition_by'] = target_data['partition_by']
            
            if target_data.get('options'):
                config['options'] = target_data['options']
            
            return config
        else:  # DB
            # Try to get db_type from frontend payload (could be db_type or database_type)
            db_type = target_data.get('db_type') or target_data.get('database_type')
            
            # If not provided, try to infer from connectionString
            conn_str = target_data.get('connectionString', '')
            if not db_type and conn_str:
                if 'oracle' in conn_str.lower():
                    db_type = 'oracle'
                elif 'mysql' in conn_str.lower():
                    db_type = 'mysql'
                elif 'sqlserver' in conn_str.lower():
                    db_type = 'sqlserver'
                    
            config = {
                "type": "DB",
                "db_type": db_type or 'postgres',  # Fallback to postgres
                "table": target_data.get('table', ''),
                "mode": target_data.get('mode', 'append')
            }
            
            # Support both connection (registry) and connectionString (direct)
            if 'connection' in target_data:
                config['connection'] = target_data['connection']
            if 'connectionString' in target_data:
                config['connectionString'] = target_data['connectionString']
                
            # Add optional DB properties
            for key in ['schema', 'primary_key', 'batchsize', 'isolationLevel']:
                if key in target_data:
                    config[key] = target_data[key]
                    
            return config
    
    @staticmethod
    def _build_audit_section(audit_data: Dict[str, Any]) -> Dict[str, Any]:
        """Build audit configuration section"""
        return {
            "enable": audit_data.get('enable', True),
            "capture_row_count": audit_data.get('capture_row_count', True),
            "capture_duration": audit_data.get('capture_duration', True),
            "path": audit_data.get('path', '/data/audit')
        }
    
    @staticmethod
    def _build_reject_section(reject_data: Dict[str, Any]) -> Dict[str, Any]:
        """Build reject handling configuration section"""
        return {
            "enable": reject_data.get('enable', True),
            "path": reject_data.get('path', '/data/reject'),
            "include_reason": reject_data.get('include_reason', True)
        }

@app.route('/api/generate-config', methods=['POST'])
def generate_config():
    """
    Generate pipeline JSON configuration from frontend data
    
    Request body: Pipeline configuration data from frontend
    Response: Generated JSON configuration
    """
    try:
        data = request.json
        
        if not data:
            return jsonify({"error": "No data provided"}), 400
        
        # Generate configuration
        config = PipelineConfigGenerator.generate_pipeline_config(data)
        
        return jsonify({
            "success": True,
            "config": config
        })
    
    except Exception as e:
        return jsonify({
            "success": False,
            "error": str(e)
        }), 500


@app.route('/api/save-config', methods=['POST'])
def save_config():
    """
    Save generated configuration to file
    
    Request body: {
        "config": {...},
        "filename": "pipeline_name.json"
    }
    """
    try:
        data = request.json
        config = data.get('config')
        filename = data.get('filename', f"pipeline_{datetime.now().strftime('%Y%m%d_%H%M%S')}.json")
        
        if not config:
            return jsonify({"error": "No configuration provided"}), 400
        
        # Ensure filename ends with .json
        if not filename.endswith('.json'):
            filename += '.json'
        
        # Save to config directory
        filepath = os.path.join(CONFIG_DIR, filename)
        with open(filepath, 'w') as f:
            json.dump(config, f, indent=2)
        
        return jsonify({
            "success": True,
            "message": f"Configuration saved to {filename}",
            "filepath": filepath
        })
    
    except Exception as e:
        return jsonify({
            "success": False,
            "error": str(e)
        }), 500

    """List all saved pipeline configurations"""
    try:

            "success": True,
            "configs": configs

    except Exception as e:
        return jsonify({
            "success": False,
            "error": str(e)
        }), 500

@app.route('/api/validate-config', methods=['POST'])
def validate_config():
    """Validate a pipeline configuration"""
    try:
        config = request.json
        
        if not config:
            return jsonify({"error": "No configuration provided"}), 400
        
        # Basic validation
        required_sections = ['pipeline', 'source', 'target']
        missing = [s for s in required_sections if s not in config]
        
        if missing:
            return jsonify({
                "valid": False,
                "errors": [f"Missing required section: {s}" for s in missing]
            })
        
        # Validate pipeline type
        valid_types = ['FILE_TO_FILE', 'FILE_TO_DB', 'DB_TO_DB', 'DB_TO_FILE']
        pipeline_type = config['pipeline'].get('pipeline_type')
        
        if pipeline_type not in valid_types:
            return jsonify({
                "valid": False,
                "errors": [f"Invalid pipeline_type: {pipeline_type}. Must be one of {valid_types}"]
            })
        
        return jsonify({
            "valid": True,
            "message": "Configuration is valid"
        })
    
    except Exception as e:
        return jsonify({
            "valid": False,
            "errors": [str(e)]
        }), 500


@app.route('/api/execute-pipeline', methods=['POST'])
def execute_pipeline():
    """
    Execute a pipeline configuration
    
    This endpoint saves the config and triggers the ETL engine execution.
    """
    try:
        config = request.json
        
        if not config:
            return jsonify({"error": "No configuration provided"}), 400
        
        # Validate configuration first
        required_sections = ['pipeline', 'source', 'target']
        missing = [s for s in required_sections if s not in config]
        
        if missing:
            return jsonify({
                "success": False,
                "error": f"Missing required sections: {missing}"
            }), 400
        
        pipeline_id = config['pipeline'].get('pipeline_id', f"PIPELINE_{datetime.now().strftime('%Y%m%d_%H%M%S')}")
        
        # Save configuration
        filename = f"{pipeline_id}.json"
        filepath = os.path.join(CONFIG_DIR, filename)
        with open(filepath, 'w') as f:
            json.dump(config, f, indent=2)
        
        # Execute the PySpark ETL engine
        import subprocess
        import time
        
        start_time = time.time()
        
        # Get paths
        backend_dir = os.path.dirname(__file__)
        etl_engine_path = os.path.join(backend_dir, 'core', 'etl_engine.py')
        
        # Check if running in virtual environment
        venv_python = os.path.join(backend_dir, '.venv', 'bin', 'python')
        if os.path.exists(venv_python):
            python_cmd = venv_python
        else:
            python_cmd = 'python3'
        
        # For development: Run with Python directly (not spark-submit)
        # In production, this should use: spark-submit etl_engine.py config.json
        stdout_log = None
        stderr_log = None
        try:
            # Find JARs
            jars_dir = os.path.join(backend_dir, 'jars')
            jar_files = []
            if os.path.exists(jars_dir):
                jar_files = [os.path.join(jars_dir, f) for f in os.listdir(jars_dir) if f.endswith('.jar')]
            
            cmd = ['spark-submit']
            
            # Add driver extra class path for both driver and executor
            if jar_files:
                # On Windows, spark-submit and Java expect forward slashes in paths.
                # spark.driver.extraClassPath uses ; as separator on Windows (OS classpath sep).
                # --jars needs comma-separated forward-slash paths.
                import sys as _sys
                is_windows = _sys.platform == 'win32'
                
                # Forward-slash paths for Spark (works on all platforms)
                fwd_jar_files = [f.replace('\\', '/') for f in jar_files]
                jars_arg = ','.join(fwd_jar_files)
                
                # OS-native classpath separator for extraClassPath
                cp_sep = ';' if is_windows else ':'
                classpath_arg = cp_sep.join(jar_files)  # OS native paths for JVM classpath
                
                cmd.extend(['--jars', jars_arg])
                # --driver-class-path is the most reliable way to get jars on driver JVM
                cmd.extend(['--driver-class-path', classpath_arg])
                cmd.extend(['--conf', f'spark.driver.extraClassPath={classpath_arg}'])
                cmd.extend(['--conf', f'spark.executor.extraClassPath={classpath_arg}'])
            
            cmd.extend([etl_engine_path, filepath])

            # Java 17+ compatibility: set JAVA_TOOL_OPTIONS in this (parent) process
            # so the JVM launched by spark-submit inherits it before it starts.
            # setdefault preserves any value the user may have already set.
            os.environ.setdefault('JAVA_TOOL_OPTIONS', (
                "--add-opens=java.base/sun.nio.ch=ALL-UNNAMED "
                "--add-opens=java.base/java.lang=ALL-UNNAMED "
                "--add-opens=java.base/java.lang.reflect=ALL-UNNAMED "
                "--add-opens=java.base/java.io=ALL-UNNAMED "
                "--add-opens=java.base/java.util=ALL-UNNAMED "
                "--add-exports=java.base/sun.nio.ch=ALL-UNNAMED "
                "--add-exports=java.base/javax.security.auth=ALL-UNNAMED"
            ))

            # Try to execute with spark-submit if available
            result = subprocess.run(
                cmd,
                capture_output=True,
                text=True,
                timeout=300  # 5 minute timeout
            )
            
            execution_successful = result.returncode == 0
            execution_output = result.stdout
            execution_error = result.stderr
            
            # Write full logs to debug files for inspection
            debug_log_dir = os.path.join(backend_dir, 'logs')
            os.makedirs(debug_log_dir, exist_ok=True)
            
            stdout_log = os.path.join(debug_log_dir, f'{pipeline_id}_stdout.log')
            stderr_log = os.path.join(debug_log_dir, f'{pipeline_id}_stderr.log')
            
            with open(stdout_log, 'w') as f:
                f.write(execution_output or '')
            with open(stderr_log, 'w') as f:
                f.write(execution_error or '')
            
        except FileNotFoundError:
            # spark-submit not found, return informative message
            execution_successful = False
            execution_output = ""
            execution_error = "spark-submit not found. PySpark is not installed or not in PATH."
        
        except subprocess.TimeoutExpired:
            execution_successful = False
            execution_output = ""
            execution_error = "Pipeline execution timed out after 5 minutes"
        
        duration = time.time() - start_time
        
        if execution_successful:
            # Parse actual metrics from PySpark output
            source_count = 0
            target_count = 0
            reject_count = 0
            cdc_insert_count = None
            cdc_update_count = None
            cdc_delete_count = None

            import re
            if execution_output:
                source_match = re.search(r'Source row count:\s*(\d+)', execution_output)
                if source_match:
                    source_count = int(source_match.group(1))

                target_match = re.search(r'Target row count:\s*(\d+)', execution_output)
                if target_match:
                    target_count = int(target_match.group(1))

                reject_match = re.search(r"'reject_row_count':\s*(\d+)", execution_output)
                if reject_match:
                    reject_count = int(reject_match.group(1))

                # CDC breakdown metrics
                ins_match = re.search(r'CDC insert row count:\s*(\d+)', execution_output)
                if ins_match:
                    cdc_insert_count = int(ins_match.group(1))

                upd_match = re.search(r'CDC update row count:\s*(\d+)', execution_output)
                if upd_match:
                    cdc_update_count = int(upd_match.group(1))

                del_match = re.search(r'CDC delete row count:\s*(\d+)', execution_output)
                if del_match:
                    cdc_delete_count = int(del_match.group(1))

            metrics = {
                "source_row_count": source_count,
                "target_row_count": target_count,
                "reject_row_count": reject_count
            }
            # Only add CDC breakdown when CDC was actually used
            if cdc_insert_count is not None:
                metrics["cdc_insert_row_count"] = cdc_insert_count
                metrics["cdc_update_row_count"] = cdc_update_count or 0
                metrics["cdc_delete_row_count"] = cdc_delete_count or 0

            result = {
                "success": True,
                "pipeline_id": pipeline_id,
                "status": "completed",
                "duration": round(duration, 2),
                "metrics": metrics,
                "message": f"Pipeline {pipeline_id} executed successfully",
                "config_path": filepath,
                "execution_log": execution_output[-10000:] if execution_output else "",
                "full_log_path": stdout_log if execution_output else None
            }
        else:
            result = {
                "success": False,
                "pipeline_id": pipeline_id,
                "status": "failed",
                "duration": round(duration, 2),
                "error": execution_error[-10000:] if execution_error else "Pipeline execution failed",  # Last 10000 chars
                "message": f"Pipeline {pipeline_id} execution failed",
                "config_path": filepath,
                "execution_log": execution_output[-10000:] if execution_output else "",  # Include stdout even on failure
                "full_log_path": stderr_log if execution_error else None,
                "note": "To run pipelines, install PySpark: pip install pyspark"
            }
        
        return jsonify(result)
    
    except Exception as e:
        return jsonify({
            "success": False,
            "error": str(e)
        }), 500


@app.route('/api/connections', methods=['GET'])
def list_connections():
    """List available connection strings from environment variables"""
    connections = []
    # Filter for env vars that look like DB connections (Full URLs only)
    patterns = ['_URL', '_CONN', '_STRING', '_JDBC']
    for key, value in os.environ.items():
        if any(pat in key for pat in patterns):
            # Mask value
            masked = f"{value[:8]}..." if len(value) > 8 else "***"
            connections.append({"key": key, "value": masked})
    return jsonify({"success": True, "connections": connections})


@app.route('/api/test-connection', methods=['POST'])
def test_connection():
    """
    Test database connection
    
    Request body: {
        "type": "postgres",
        "connectionString": "postgresql://user:pass@host:port/db",
        "schema": "public",
        "table": "employees"
    }
    
    Returns: {
        "success": true/false,
        "error": "error message if failed"
    }
    """
    try:
        data = request.json
        db_type = data.get('type')
        connection_string = data.get('connectionString')
        
        if not db_type:
            return jsonify({"success": False, "error": "Database type is required"}), 400
        
        if not connection_string:
            return jsonify({"success": False, "error": "Connection string is required"}), 400
            
        # Check if connection_string is an environment variable key
        if connection_string in os.environ:
            connection_string = os.environ[connection_string]
        
        # Import required modules
        from urllib.parse import urlparse, unquote
        
        # Remove jdbc: prefix if present
        if connection_string.startswith("jdbc:"):
            connection_string = connection_string[5:]
            
        # Parse connection string
        parsed = urlparse(connection_string)
        
        # Validate that it looks like a URL (must have scheme or netloc)
        if not parsed.scheme or not parsed.netloc:
            # Fallback checks?
            if db_type == 'postgres' and "=" in connection_string:
                 # Allow DSN style for postgres? 
                 pass
            else:
                 return jsonify({"success": False, "error": "Invalid connection string format. Expected URL (e.g. postgresql://user:pass@host:port/db)"})

        username = unquote(parsed.username) if parsed.username else ''
        password = unquote(parsed.password) if parsed.password else ''
        hostname = parsed.hostname or 'localhost'
        port = parsed.port
        database = parsed.path.lstrip('/') if parsed.path else ''
        
        # Ensure database is not empty if required
        if not database and db_type in ['postgres', 'mysql', 'oracle', 'sqlserver'] and parsed.scheme:
             # handle case where path is empty
             pass
        
        # Test connection based on database type
        if db_type == 'postgres':
            try:
                import psycopg2
                conn = psycopg2.connect(
                    host=hostname,
                    port=port or 5432,
                    database=database,
                    user=username,
                    password=password,
                    connect_timeout=5
                )
                conn.close()
                return jsonify({"success": True})
            except ImportError:
                return jsonify({
                    "success": False,
                    "error": "PostgreSQL driver not installed. Run: pip install psycopg2-binary"
                })
            except Exception as e:
                return jsonify({"success": False, "error": str(e)})
                
        elif db_type == 'mysql':
            try:
                import pymysql
                conn = pymysql.connect(
                    host=hostname,
                    port=port or 3306,
                    database=database,
                    user=username,
                    password=password,
                    connect_timeout=5
                )
                conn.close()
                return jsonify({"success": True})
            except ImportError:
                return jsonify({
                    "success": False,
                    "error": "MySQL driver not installed. Run: pip install pymysql"
                })
            except Exception as e:

@app.route('/api/preview/file-headers', methods=['POST'])
def get_file_headers():
    """
    Get column headers from a file
    
    Request body (format 1 - frontend format):
    {
        "type": "csv",
        "path": "/path/to/file.csv",
        "delimiter": ","
    }
    
    OR (format 2 - legacy format):
    {
        "path": "/path/to/file.csv",
        "format": "csv",
        "options": {"header": true, "delimiter": ","}
    }
    
    Returns: {
        "headers": ["col1", "col2", ...]
    }
    """
    try:
        data = request.json
        
        # Support both frontend format and legacy format
        file_path = data.get('path')
        file_format = data.get('format') or data.get('type', 'csv')
        
        # Get delimiter from either location
        delimiter = data.get('delimiter')
        if not delimiter and 'options' in data:
            delimiter = data['options'].get('delimiter', ',')
        if not delimiter:
            delimiter = ','
        
        # Get header option
        options = data.get('options', {})
        has_header = options.get('header', True)
        
        if not file_path:
            return jsonify({"error": "File path is required"}), 400
        
        if not os.path.exists(file_path):
            return jsonify({"error": f"File not found: {file_path}"}), 404
        
        headers = []
        
        if file_format in ['csv', 'txt', 'dat']:
            import csv
            
            with open(file_path, 'r') as f:
                reader = csv.reader(f, delimiter=delimiter)
                first_row = next(reader, None)
                
                if first_row:
                    if has_header:
                        headers = first_row
                    else:
                        # Generate column names like col0, col1, etc.
                        headers = [f"col{i}" for i in range(len(first_row))]
        
        elif file_format == 'json':
            with open(file_path, 'r') as f:
                file_data = json.load(f)
                if isinstance(file_data, list) and len(file_data) > 0:
                    headers = list(file_data[0].keys())
                elif isinstance(file_data, dict):
                    headers = list(file_data.keys())
        
        elif file_format in ['parquet', 'orc', 'avro', 'excel']:
            # For these formats, we'd need PySpark or pyarrow
            # For now, return a message
            return jsonify({
                "error": f"Preview for {file_format} format requires PySpark. Please use manual column names."
            }), 501
        
        else:
            return jsonify({"error": f"Unsupported format: {file_format}"}), 400
        
        return jsonify({"headers": headers})
    
    except Exception as e:
        return jsonify({"error": str(e)}), 500


@app.route('/api/preview/db-columns', methods=['POST'])
def get_db_columns():
    """
    Get column names from a database table
    
    Request body: {
        "db_type": "postgres",
        "table": "employees",
        "schema": "public",
        "connectionString": "postgresql://user:pass@host:port/db"
        OR
        "connection": "pg_conn"  (for pre-registered connections)
    }
    
    Returns: {
        "columns": ["col1", "col2", ...]
    }
    """
    try:
        data = request.json
        db_type = data.get('db_type')
        table = data.get('table')
        schema = data.get('schema', 'public')
        connection = data.get('connection')
        connection_string = data.get('connectionString')
        
        if not all([db_type, table]):
            return jsonify({"error": "db_type and table are required"}), 400
        
        if not connection and not connection_string:
            return jsonify({"error": "Either connection or connectionString is required"}), 400
        
        # Import required modules
        from urllib.parse import urlparse, unquote
        # Drivers will be imported based on db_type
        
        columns = []
        
        # Parse connection string if provided
        if connection_string:
            # Check if connection_string is an environment variable key
            if connection_string in os.environ:
                connection_string = os.environ[connection_string]
            
            # Remove jdbc: prefix if present
            if connection_string.startswith("jdbc:"):
                connection_string = connection_string[5:]
                
            parsed = urlparse(connection_string)
            username = unquote(parsed.username) if parsed.username else ''
            password = unquote(parsed.password) if parsed.password else ''
            hostname = parsed.hostname or 'localhost'
            port = parsed.port
            database = parsed.path.lstrip('/') if parsed.path else ''
            
            # Connect based on database type
            if db_type == 'postgres':
                import psycopg2
                conn = psycopg2.connect(
                    host=hostname,
                    port=port or 5432,
                    database=database,
                    user=username,
                    password=password
                )
                cursor = conn.cursor()
                
                # Query to get column names
                qualified_table = f"{schema}.{table}" if schema else table
                cursor.execute(f"""
                    SELECT column_name 
                    FROM information_schema.columns 
                    WHERE table_schema = %s AND table_name = %s
                    ORDER BY ordinal_position
                """, (schema, table))
                
                columns = [row[0] for row in cursor.fetchall()]
                cursor.close()
                conn.close()
                
            elif db_type == 'mysql':
                import pymysql
                conn = pymysql.connect(
                    host=hostname,
                    port=port or 3306,
                    database=database,
                    user=username,
                    password=password
                )
                cursor = conn.cursor()
                
                # Query to get column names
                cursor.execute(f"""
                    SELECT column_name 
                    FROM information_schema.columns 
                    WHERE table_schema = %s AND table_name = %s
                    ORDER BY ordinal_position
                """, (database, table))
                
                columns = [row[0] for row in cursor.fetchall()]
                cursor.close()
                conn.close()
                
            elif db_type == 'oracle':
                import oracledb
                o_user, o_pwd, o_host, o_port, o_svc = _parse_oracle_connection(connection_string)
                conn = oracledb.connect(
                    user=o_user, password=o_pwd,
                    host=o_host, port=o_port,
                    service_name=o_svc
                )
                cursor = conn.cursor()

                # Oracle stores object names in UPPERCASE by default
                cursor.execute("""
                    SELECT COLUMN_NAME
                    FROM ALL_TAB_COLUMNS
                    WHERE UPPER(OWNER) = UPPER(:owner)
                      AND UPPER(TABLE_NAME) = UPPER(:table_name)
                    ORDER BY COLUMN_ID
                """, owner=schema or o_user, table_name=table)

                columns = [row[0] for row in cursor.fetchall()]
                cursor.close()
                conn.close()

            else:
                return jsonify({
                    "error": f"Database type '{db_type}' not yet supported for column preview",
                    "supported": ["postgres", "mysql", "oracle"]
                }), 400
        else:
            # Using pre-registered connection (not implemented in this endpoint yet)
            return jsonify({
                "error": "Pre-registered connections not supported in column preview yet. Please use connectionString."
            }), 400
        
        if not columns:
            return jsonify({
                "error": f"Table '{table}' not found or has no columns"
            }), 404
        
        return jsonify({"columns": columns})
    
    except ImportError as e:
        # Database drivers not installed - return helpful error with mock data
        return jsonify({
            "error": f"Database driver not installed: {str(e)}",
            "hint": "Install required package: pip install psycopg2-binary pymysql oracledb",
            "columns": ["id", "name", "email", "created_at", "updated_at"],  # Mock data as fallback
            "note": "Showing mock columns - install database drivers for real column names"
        }), 200  # Return 200 so UI doesn't break
    except Exception as e:
        return jsonify({"error": f"Failed to retrieve columns: {str(e)}"}), 500


@app.route('/api/fs/list', methods=['GET'])
def list_directory():
    """
    List files and directories in a given path
    
    Query params:
        path: Directory path to list (optional, defaults to user home or current directory)
    
    Returns:
        {
            "current_path": "/absolute/path",
            "parent_path": "/parent/path" or null,
            "items": [
                {
                    "name": "filename.txt",
                    "path": "/absolute/path/filename.txt",
                    "is_dir": false,
                    "size": 1024,
                    "modified": "2024-01-01T12:00:00"
                }
            ]
        }
    """
    try:
        # Get requested path from query params
        requested_path = request.args.get('path', '')
        
        # Security: Resolve to absolute path and prevent directory traversal
        if not requested_path:
            # Default to user home directory
            current_path = os.path.expanduser('~')
        else:
            # Resolve to absolute path
            current_path = os.path.abspath(os.path.expanduser(requested_path))
        
        # Check if path exists and is a directory
        if not os.path.exists(current_path):
            return jsonify({
                "error": f"Path does not exist: {current_path}"
            }), 404
        
        if not os.path.isdir(current_path):
            return jsonify({
                "error": f"Path is not a directory: {current_path}"
            }), 400
        
        # Get parent directory
        parent_path = os.path.dirname(current_path)
        # If we're at root, parent is None
        if parent_path == current_path:
            parent_path = None
        
        # List directory contents
        items = []
        try:
            entries = os.listdir(current_path)
            
            for entry_name in sorted(entries):
                entry_path = os.path.join(current_path, entry_name)
                
                try:
                    # Get file stats
                    stat_info = os.stat(entry_path)
                    is_dir = os.path.isdir(entry_path)
                    
                    items.append({
                        "name": entry_name,
                        "path": entry_path,
                        "is_dir": is_dir,
                        "size": stat_info.st_size if not is_dir else 0,
                        "modified": datetime.fromtimestamp(stat_info.st_mtime).isoformat()
                    })
                except (OSError, PermissionError):
                    # Skip files we can't access
                    continue
        
        except PermissionError:
            return jsonify({
                "error": f"Permission denied: {current_path}"
            }), 403
        
        return jsonify({
            "current_path": current_path,
            "parent_path": parent_path,
            "items": items
        })
    
    except Exception as e:
        return jsonify({
            "error": str(e)
        }), 500
