#!/usr/bin/env python3
"""
get_count_csv.py - Count occurrences of values in a CSV column

Usage:
    python3 get_count_csv.py -i input.csv -c column_name
    python3 get_count_csv.py -i input.csv -c column_name -o output.csv
"""
import csv
import argparse
from collections import Counter
import sys

def count_column_values(input_file, column_name, output_file=None):
    """
    Count occurrences of values in a specified column.
    
    Args:
        input_file: Path to input CSV file
        column_name: Name of the column to count
        output_file: Optional path to output CSV file (None = stdout)
    """
    try:
        with open(input_file, 'r') as f:
            reader = csv.DictReader(f)
            
            # Check if column exists
            if column_name not in reader.fieldnames:
                print(f"Error: Column '{column_name}' not found in CSV", file=sys.stderr)
                print(f"Available columns: {', '.join(reader.fieldnames)}", file=sys.stderr)
                return
            
            # Count values in the specified column
            counter = Counter()
            total_rows = 0
            
            for row in reader:
                value = row[column_name].strip()
                if value:  # Only count non-empty values
                    counter[value] += 1
                total_rows += 1
            
            # Sort by count (descending) then by value (ascending)
            sorted_counts = sorted(counter.items(), key=lambda x: (-x[1], x[0]))
            
            # Output results
            if output_file:
                with open(output_file, 'w', newline='') as out:
                    writer = csv.writer(out)
                    writer.writerow([column_name, 'Count'])
                    writer.writerows(sorted_counts)
                print(f"Results written to {output_file}", file=sys.stderr)
            else:
                print(f"\nColumn: {column_name}")
                print(f"Total rows: {total_rows}")
                print(f"Unique values: {len(counter)}")
                print(f"\n{'Value':<40} {'Count':>10}")
                print("-" * 52)
                for value, count in sorted_counts:
                    print(f"{value:<40} {count:>10}")
    
    except FileNotFoundError:
        print(f"Error: File '{input_file}' not found", file=sys.stderr)
    except Exception as e:
        print(f"Error: {e}", file=sys.stderr)

if __name__ == "__main__":
    parser = argparse.ArgumentParser(
        description="Count occurrences of values in a CSV column",
        formatter_class=argparse.RawDescriptionHelpFormatter,
        epilog="""
Examples:
  python3 get_count_csv.py -i data.csv -c status
  python3 get_count_csv.py -i data.csv -c type -o counts.csv
        """
    )
    parser.add_argument('-i', '--input', required=True, help='Input CSV file (required)')
    parser.add_argument('-c', '--column', required=True, help='Column name to count (required)')
    parser.add_argument('-o', '--output', default=None, help='Output CSV file (optional, default: stdout)')
    
    args = parser.parse_args()
    
    count_column_values(args.input, args.column, args.output)
