#!/bin/bash
#
# get_files_directory_list.sh - List all directories, subdirectories, and files with deployment dates
#
# Usage:
#   ./get_files_directory_list.sh /path/to/directory
#   ./get_files_directory_list.sh /path/to/directory output.csv
#

# Check if path argument is provided
if [ $# -lt 1 ]; then
    echo "Usage: $0 <directory_path> [output_file.csv]" >&2
    echo "Example: $0 /var/www/html" >&2
    echo "Example: $0 /var/www/html output.csv" >&2
    exit 1
fi

TARGET_PATH="$1"
OUTPUT_FILE="${2:-}"

# Check if target path exists
if [ ! -d "$TARGET_PATH" ]; then
    echo "Error: Directory '$TARGET_PATH' does not exist" >&2
    exit 1
fi

# Function to output results
output_results() {
    if [ -n "$OUTPUT_FILE" ]; then
        # Write header
        echo "Type,Path,Name,Modified_Date,Size_Bytes,Permissions,Owner" > "$OUTPUT_FILE"
        
        # Find all files and directories, format output
        find "$TARGET_PATH" -print0 | while IFS= read -r -d '' item; do
            if [ -d "$item" ]; then
                TYPE="Directory"
            else
                TYPE="File"
            fi
            
            # Get file stats
            MODIFIED_DATE=$(stat -f "%Sm" -t "%Y-%m-%d %H:%M:%S" "$item" 2>/dev/null || stat -c "%y" "$item" 2>/dev/null | cut -d'.' -f1)
            SIZE=$(stat -f "%z" "$item" 2>/dev/null || stat -c "%s" "$item" 2>/dev/null)
            PERMS=$(stat -f "%Sp" "$item" 2>/dev/null || stat -c "%A" "$item" 2>/dev/null)
            OWNER=$(stat -f "%Su" "$item" 2>/dev/null || stat -c "%U" "$item" 2>/dev/null)
            NAME=$(basename "$item")
            
            echo "$TYPE,\"$item\",\"$NAME\",\"$MODIFIED_DATE\",$SIZE,$PERMS,$OWNER"
        done >> "$OUTPUT_FILE"
        
        echo "Results written to $OUTPUT_FILE" >&2
        echo "Total items: $(tail -n +2 "$OUTPUT_FILE" | wc -l)" >&2
    else
        # Output to stdout with formatted table
        echo "Scanning: $TARGET_PATH" >&2
        echo "" >&2
        printf "%-12s %-20s %-80s %-10s %s\n" "TYPE" "MODIFIED DATE" "PATH" "SIZE" "OWNER"
        printf "%s\n" "$(printf '%.0s-' {1..150})"
        
        find "$TARGET_PATH" -print0 | while IFS= read -r -d '' item; do
            if [ -d "$item" ]; then
                TYPE="Directory"
            else
                TYPE="File"
            fi
            
            MODIFIED_DATE=$(stat -f "%Sm" -t "%Y-%m-%d %H:%M:%S" "$item" 2>/dev/null || stat -c "%y" "$item" 2>/dev/null | cut -d'.' -f1)
            SIZE=$(stat -f "%z" "$item" 2>/dev/null || stat -c "%s" "$item" 2>/dev/null)
            OWNER=$(stat -f "%Su" "$item" 2>/dev/null || stat -c "%U" "$item" 2>/dev/null)
            
            printf "%-12s %-20s %-80s %-10s %s\n" "$TYPE" "$MODIFIED_DATE" "$item" "$SIZE" "$OWNER"
        done
    fi
}

# Execute
output_results
