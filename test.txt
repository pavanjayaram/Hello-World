
        else:  # DB
            # Try to get db_type from frontend payload (could be db_type or database_type)
            db_type = target_data.get('db_type') or target_data.get('database_type')
            
            # If not provided, try to infer from connectionString
            conn_str = target_data.get('connectionString', '')
            if not db_type and conn_str:
                if 'oracle' in conn_str.lower():
                    db_type = 'oracle'
                elif 'mysql' in conn_str.lower():
                    db_type = 'mysql'
                elif 'sqlserver' in conn_str.lower():
                    db_type = 'sqlserver'
                    
            config = {
                "type": "DB",
                "db_type": db_type or 'postgres',  # Fallback to postgres
                "table": target_data.get('table', ''),
                "mode": target_data.get('mode', 'append')
            }
