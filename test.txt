import React, { useState } from 'react';
import { TransformationConfig, TransformationSpec } from '../types';

interface CodeExportProps {
  config: TransformationConfig;
  spec: TransformationSpec | null;
  generatedCode: string;
}

const CodeExport: React.FC<CodeExportProps> = ({ config, generatedCode }) => {
  const [copySuccess, setCopySuccess] = useState<string | null>(null);
  const [downloadSuccess, setDownloadSuccess] = useState<string | null>(null);

  const code = generatedCode;

  const generateFilename = (): string => {
    // Generate filename with timestamp and source info
    const timestamp = new Date().toISOString()
      .slice(0, 19)
      .replace(/:/g, '-')
      .replace('T', '_');

    const sourceType = config.dataSource.source?.type || 'unknown';
    const targetType = config.dataSource.target?.type || 'unknown';

    return `transformation_${sourceType}_to_${targetType}_${timestamp}.py`;
  };

  const copyToClipboard = async () => {
    try {
      await navigator.clipboard.writeText(code);
      setCopySuccess('Code copied to clipboard!');

      // Clear success message after 3 seconds
      setTimeout(() => setCopySuccess(null), 3000);
    } catch (err) {
      // Fallback for browsers that don't support clipboard API
      try {
        const textArea = document.createElement('textarea');
        textArea.value = code;
        textArea.style.position = 'fixed';
        textArea.style.left = '-999999px';
        textArea.style.top = '-999999px';
        document.body.appendChild(textArea);
        textArea.focus();
        textArea.select();

        const successful = document.execCommand('copy');
        document.body.removeChild(textArea);

        if (successful) {
          setCopySuccess('Code copied to clipboard!');
          setTimeout(() => setCopySuccess(null), 3000);
        } else {
          setCopySuccess('Failed to copy code');
          setTimeout(() => setCopySuccess(null), 3000);
        }
      } catch (fallbackErr) {
        setCopySuccess('Copy not supported in this browser');
        setTimeout(() => setCopySuccess(null), 3000);
      }
    }
  };

  const downloadCode = () => {
    try {
      const blob = new Blob([code], { type: 'text/x-python' });
      const url = URL.createObjectURL(blob);
      const a = document.createElement('a');

      a.href = url;
      a.download = generateFilename();
      a.style.display = 'none';

      document.body.appendChild(a);
      a.click();
      document.body.removeChild(a);

      // Clean up the URL object
      URL.revokeObjectURL(url);

      setDownloadSuccess(`Downloaded as ${generateFilename()}`);
      setTimeout(() => setDownloadSuccess(null), 3000);
    } catch (err) {
      setDownloadSuccess('Failed to download file');
      setTimeout(() => setDownloadSuccess(null), 3000);
    }
  };

  const getCodeStats = () => {
    const lines = code.split('\n').length;
    const characters = code.length;
    const words = code.split(/\s+/).filter(word => word.length > 0).length;
    const size = new Blob([code]).size;

    return { lines, characters, words, size };
  };

  const stats = getCodeStats();

  if (!code || code.trim() === '') {
    return (
      <div className="code-export">
        <div className="no-code-message">
          <p>No code available to export. Please generate code first.</p>
        </div>
      </div>
    );
  }

  return (
    <div className="code-export">
      <div className="code-export-header">
        <h3>Export Generated Code</h3>
        <div className="code-stats">
          <span>{stats.lines} lines</span>
          <span>{stats.characters} characters</span>
          <span>{(stats.size / 1024).toFixed(1)} KB</span>
        </div>
      </div>

      <div className="export-actions">
        <div className="export-buttons">
          <button
            onClick={copyToClipboard}
            className="btn btn-primary export-btn"
            title="Copy code to clipboard"
          >
            <span className="btn-icon">📋</span>
            Copy to Clipboard
          </button>

          <button
            onClick={downloadCode}
            className="btn btn-secondary export-btn"
            title="Download code as Python file"
          >
            <span className="btn-icon">💾</span>
            Download .py File
          </button>
        </div>

        <div className="filename-preview">
          <label>Download filename:</label>
          <code>{generateFilename()}</code>
        </div>
      </div>

      {/* Success Messages */}
      {copySuccess && (
        <div className="success-message copy-success">
          <span className="success-icon">✅</span>
          {copySuccess}
        </div>
      )}

      {downloadSuccess && (
        <div className="success-message download-success">
          <span className="success-icon">✅</span>
          {downloadSuccess}
        </div>
      )}

      {/* Code Preview */}
      <div className="code-preview">
        <h4>Code Preview</h4>
        <div className="code-preview-content">
          <pre>
            <code>
              {code.split('\n').slice(0, 10).join('\n')}
              {code.split('\n').length > 10 && '\n... (truncated)'}
            </code>
          </pre>
        </div>
      </div>

      {/* Export Options */}
      <div className="export-options">
        <h4>Export Options</h4>
        <div className="options-grid">
          <div className="option-item">
            <h5>Clipboard Copy</h5>
            <p>Copy the entire code to your system clipboard for pasting into your IDE or text editor.</p>
            <ul>
              <li>Preserves formatting and indentation</li>
              <li>Ready to paste into any text editor</li>
              <li>Works with most modern browsers</li>
            </ul>
          </div>

          <div className="option-item">
            <h5>File Download</h5>
            <p>Download the code as a Python (.py) file with a descriptive filename including timestamp.</p>
            <ul>
              <li>Automatic filename with timestamp</li>
              <li>Ready to run with spark-submit</li>
              <li>Preserves all code formatting</li>
              <li>Includes UTF-8 encoding</li>
            </ul>
          </div>
        </div>
      </div>

      {/* Usage Instructions */}
      <div className="usage-instructions">
        <h4>Next Steps</h4>
        <div className="instructions-content">
          <h5>After Copying/Downloading:</h5>
          <ol>
            <li><strong>Review the code:</strong> Check connection strings, file paths, and configuration parameters</li>
            <li><strong>Update credentials:</strong> Replace placeholder values with your actual database credentials</li>
            <li><strong>Test with sample data:</strong> Run the code on a small dataset first</li>
            <li><strong>Deploy to production:</strong> Use spark-submit or your preferred Spark deployment method</li>
          </ol>

          <h5>Execution Commands:</h5>
          <div className="command-examples">
            <div className="command-item">
              <label>Local Spark:</label>
              <code>spark-submit {generateFilename()}</code>
            </div>
            <div className="command-item">
              <label>Cluster Mode:</label>
              <code>spark-submit --master yarn --deploy-mode cluster {generateFilename()}</code>
            </div>
            <div className="command-item">
              <label>With Configuration:</label>
              <code>spark-submit --conf spark.sql.adaptive.enabled=true {generateFilename()}</code>
            </div>
          </div>
        </div>
      </div>

      {/* File Information */}
      <div className="file-info">
        <h4>File Information</h4>
        <div className="info-grid">
          <div className="info-item">
            <label>File Type:</label>
            <span>Python Script (.py)</span>
          </div>
          <div className="info-item">
            <label>Encoding:</label>
            <span>UTF-8</span>
          </div>
          <div className="info-item">
            <label>Framework:</label>
            <span>Apache Spark / PySpark</span>
          </div>
          <div className="info-item">
            <label>Generated:</label>
            <span>{new Date().toLocaleString()}</span>
          </div>
        </div>
      </div>
    </div>
  );
};

export default CodeExport;
