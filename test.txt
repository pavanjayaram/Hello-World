import React, { useState } from 'react';
import DataSourceConfigComponent from './components/DataSourceConfig';
import TransformationBuilder from './components/TransformationBuilder';
import ExecutePipeline from './components/ExecutePipeline';
import CodeDisplay from './components/CodeDisplay';
import { TransformationConfig, DataSourceConfig, FieldTransformation, RowTransformation } from './types';
import './components/styles.css';

const App: React.FC = () => {
  console.log('App component rendering at:', new Date().toISOString());

  const [currentStep, setCurrentStep] = useState(1);
  const [config, setConfig] = useState<TransformationConfig>({
    dataSource: {
      mode: 'file-to-file',
      source: {
        type: 'csv',
        delimiter: ',',
        path: ''
      },
      target: {
        type: 'csv',
        delimiter: ',',
        path: ''
      }
    },
    fieldTransforms: [],
    rowTransforms: []
  });

  const [spec, setSpec] = useState<any>(null);

  const updateDataSource = (dataSource: DataSourceConfig) => {
    setConfig(prev => ({ ...prev, dataSource }));
  };

  const updateFieldTransforms = (fieldTransforms: FieldTransformation[]) => {
    console.log('App: updateFieldTransforms called with:', fieldTransforms.map(f => f.fieldName));
    setConfig(prev => ({ ...prev, fieldTransforms }));
  };

  const updateRowTransforms = (rowTransforms: RowTransformation[]) => {
    setConfig(prev => ({ ...prev, rowTransforms }));
  };

  const steps = [
    { id: 1, title: 'Data Source Configuration', component: 'datasource' },
    { id: 2, title: 'Transformation Builder', component: 'transformations' },
    { id: 3, title: 'Review & Execute Pipeline', component: 'execute' },
    { id: 4, title: 'AI Code Generation', component: 'codedisplay' }
  ];

  const renderCurrentStep = () => {
    switch (currentStep) {
      case 1:
        return (
          <DataSourceConfigComponent
            config={config.dataSource}
            onChange={updateDataSource}
          />
        );
      case 2:
        return (
          <TransformationBuilder
            config={config.dataSource}
            fieldTransforms={config.fieldTransforms}
            rowTransforms={config.rowTransforms}
            onFieldTransformsChange={updateFieldTransforms}
            onRowTransformsChange={updateRowTransforms}
          />
        );
      case 3:
        return <ExecutePipeline config={config} />;
      case 4:
        return (
          <CodeDisplay
            config={config}
            spec={spec}
            onSpecChange={setSpec}
          />
        );
      default:
        return <div>Unknown step</div>;
    }
  };

  return (
    <div className="app">
      <header className="app-header">
        <h1>Config-Driven PySpark ETL Engine</h1>
        <p>Design your data pipeline and execute it with PySpark</p>
      </header>

      <nav className="step-navigation">
        <div className="steps">
          {steps.map((step) => (
            <button
              key={step.id}
              className={`step ${currentStep === step.id ? 'active' : ''} enabled`}
              onClick={() => setCurrentStep(step.id)}
            >
              <span className="step-number">{step.id}</span>
              <span className="step-title">{step.title}</span>
            </button>
          ))}
        </div>
      </nav>

      <main className="app-content">
        <div className="step-content">
          <h2>{steps.find(s => s.id === currentStep)?.title}</h2>
          {renderCurrentStep()}
        </div>

        <div className="step-controls">
          {currentStep > 1 && (
            <button
              className="btn btn-secondary"
              onClick={() => setCurrentStep(currentStep - 1)}
            >
              Previous
            </button>
          )}

          {currentStep < steps.length && (
            <button
              className="btn btn-primary"
              onClick={() => setCurrentStep(currentStep + 1)}
            >
              Next
            </button>
          )}
        </div>
      </main>

      <footer className="app-footer">
        <p>Config-Driven PySpark ETL Engine - "Give me a JSON, I'll run your pipeline."</p>
      </footer>
    </div>
  );
};

export default App;
