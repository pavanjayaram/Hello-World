
        else:  # DB
            config = {
                "type": "DB",
                "db_type": source_data.get('db_type', 'postgres'),
                "connection": source_data.get('connection', 'pg_conn')
            }
            if 'query' in source_data and source_data['query']:
                config['query'] = source_data['query']
            else:
                config['table'] = source_data.get('table', '')
            return config
